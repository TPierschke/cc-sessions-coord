# cc-sessions-coord Implementation Plan (.NET 8)

> **For agentic workers:** Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement task-by-task. Steps use checkbox `- [ ]` syntax for tracking.

**Goal:** NSSM-Service `cc-sessions-coord` ersetzt fragiles Hook→Claude→MCP-Sync und SQLite-Coord durch HTTP-API-Worker (.NET 8) mit Postgres-Backbone (`cc_sessions_coord` DB).

**Architecture:** Worker mit ASP.NET Core Minimal API auf Kestrel (`localhost:7733`) ist Single Writer in Postgres. Sessions schreiben via HTTP, niemals direkt. Hooks haben 1000ms Timeout + Circuit-Breaker, fire-and-forget. Sink-Sync (Hindsight+Shodh) läuft als 30s-`BackgroundService` mit Lease-Mechanik und Fehler-Klassifizierung. Sink-Idempotenz live verifiziert (siehe `docs/sink-idempotency-2026-05-07.md`) — Reconciliation-Loop ist optionales Phase-2-Safety-Net.

**Tech Stack:** .NET 8 (`Microsoft.NET.Sdk.Web`), Kestrel, Minimal API, Npgsql 10.0.2, Swashbuckle 6.6.2, Microsoft.AspNetCore.OpenApi 8.0.25, NSSM. Tests: xUnit 2.5.3 + coverlet 6.0.0 + `Microsoft.AspNetCore.Mvc.Testing` (`WebApplicationFactory`) + `WireMock.Net` für HTTP-Mocks. Postgres-Tests laufen gegen lokale `cc_sessions_coord_test`-DB (kein Testcontainers — User-Präferenz: Live-DB, kein Container-Overhead).

**Spec:** `docs/spec/2026-05-07-design.md` (status: final)

public sealed class NpgsqlConnectionFactory
{
    private readonly string _connString;
    public NpgsqlConnectionFactory(IConfiguration cfg) =>
        _connString = cfg.GetConnectionString("Default")
            ?? throw new InvalidOperationException("ConnectionStrings:Default missing");
    public async Task<NpgsqlConnection> OpenAsync(CancellationToken ct = default)
    {
        var conn = new NpgsqlConnection(_connString);
        await conn.OpenAsync(ct);
        return conn;
    }
}

public sealed class WorkerStateRepository
{
    private readonly NpgsqlConnectionFactory _factory;
    public WorkerStateRepository(NpgsqlConnectionFactory factory) => _factory = factory;

    public async Task<WorkerLease> AcquireLeaseAsync(CancellationToken ct = default)
    {
        await using var conn = await _factory.OpenAsync(ct);
        await using var cmd = conn.CreateCommand();
        cmd.CommandText = """
            WITH new_token AS (SELECT nextval('memory_sync.fencing_seq') AS tok)
            INSERT INTO memory_sync.worker_state (id, worker_epoch, fencing_token, started_at, last_heartbeat)
            SELECT 1, @epoch, tok, now(), now() FROM new_token
            ON CONFLICT (id) DO UPDATE
                SET worker_epoch = EXCLUDED.worker_epoch,
                    fencing_token = EXCLUDED.fencing_token,
                    started_at = now(),
                    last_heartbeat = now()
            RETURNING worker_epoch, fencing_token, started_at;
            """;
        var epoch = Guid.NewGuid();
        cmd.Parameters.AddWithValue("epoch", epoch);
        await using var r = await cmd.ExecuteReaderAsync(ct);
        await r.ReadAsync(ct);
        return new WorkerLease(r.GetGuid(0), r.GetInt64(1), r.GetFieldValue<DateTimeOffset>(2));
    }

    public async Task HeartbeatAsync(Guid epoch, CancellationToken ct = default) { /* UPDATE last_heartbeat */ }
}

```csharp
public interface IShodhClient
{
    Task<SinkResult> SendAsync(SyncEventRow row, CancellationToken ct);
}

public sealed class ShodhClient : IShodhClient
{
    private readonly HttpClient _http;
    private readonly ShodhOptions _opts;
    public ShodhClient(HttpClient http, IOptions<ShodhOptions> opts) { _http = http; _opts = opts.Value; }

    public async Task<SinkResult> SendAsync(SyncEventRow row, CancellationToken ct)
    {
        // Em-Dash defensiv ersetzen (Parser-Bug, siehe sink-idempotency-Doku)
        var content = row.Content.Replace('—', '-');
        if (content.Length < 10) content = content.PadRight(10, ' ');
        var body = new {
            user_id = _opts.UserId,
            content,
            event_id = row.EventId.ToString()  // Tracking, nicht Idempotenz-Key
        };
        using var req = new HttpRequestMessage(HttpMethod.Post, "/api/remember") {
            Content = JsonContent.Create(body)
        };
        req.Headers.Add("X-API-Key", _opts.ApiKey);
        try {
            using var resp = await _http.SendAsync(req, ct);
            return new SinkResult(ErrorClassifier.ClassifyHttp((int)resp.StatusCode),
                                  (int)resp.StatusCode, await resp.Content.ReadAsStringAsync(ct));
        } catch (Exception ex) {
            return new SinkResult(ErrorClassifier.ClassifyException(ex), 0, ex.Message);
        }
    }
}

public sealed class ShodhOptions { public string Url { get; set; } = ""; public string ApiKey { get; set; } = ""; public string UserId { get; set; } = ""; }
public sealed record SinkResult(SinkStatus Status, int HttpCode, string? Detail);
```

public sealed class HindsightClient : IHindsightClient
{
    public async Task<SinkResult> SendAsync(SyncEventRow row, CancellationToken ct)
    {
        var rpc = new {
            jsonrpc = "2.0",
            id = 1,
            method = "tools/call",
            @params = new {
                name = "retain",
                arguments = new {
                    content = row.Content,
                    context = _opts.Context,
                    document_id = row.EventId.ToString(),
                    tags = new[] { row.MemoryType ?? "memory" },
                    metadata = new Dictionary<string, string?> {
                        ["source"] = row.Source,
                        ["session_id"] = row.SessionId,
                        ["memory_path"] = row.MemoryPath
                    }
                }
            }
        };
        using var req = new HttpRequestMessage(HttpMethod.Post, _opts.Url) {
            Content = JsonContent.Create(rpc)
        };
        req.Headers.Accept.ParseAdd("text/event-stream, application/json");
        try {
            using var resp = await _http.SendAsync(req, HttpCompletionOption.ResponseHeadersRead, ct);
            var raw = await resp.Content.ReadAsStringAsync(ct);
            // SSE-Body parsen: erste data:-Zeile bis "\n\n"
            var dataLine = ExtractDataLine(raw);
            var json = JsonSerializer.Deserialize<JsonElement>(dataLine);
            if (json.TryGetProperty("error", out _))
                return new SinkResult(SinkStatus.ErrorPermanent, (int)resp.StatusCode, dataLine);
            return new SinkResult(ErrorClassifier.ClassifyHttp((int)resp.StatusCode),
                                  (int)resp.StatusCode, dataLine);
        } catch (Exception ex) {
            return new SinkResult(ErrorClassifier.ClassifyException(ex), 0, ex.Message);
        }
    }

    private static string ExtractDataLine(string sseBody)
    {
        // Lines: "event: message\ndata: {...}\n\n" — gib alles nach "data: " bis "\n\n" zurück
        var i = sseBody.IndexOf("data: ", StringComparison.Ordinal);
        if (i < 0) return sseBody;
        var start = i + "data: ".Length;
        var end = sseBody.IndexOf("\n\n", start, StringComparison.Ordinal);
        return end < 0 ? sseBody[start..].Trim() : sseBody[start..end];
    }
}

```csharp
public sealed class SinkSyncService : BackgroundService
{
    private readonly SyncQueueRepository _queue;
    private readonly IHindsightClient _hindsight;
    private readonly IShodhClient _shodh;
    private readonly WorkerLeaseHolder _lease;  // injected, gehalten von SingleInstanceLockService
    private readonly ILogger<SinkSyncService> _log;

    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            try { await RunOneCycleAsync(ct); }
            catch (Exception ex) { _log.LogError(ex, "sink-sync cycle failed"); }
            await Task.Delay(TimeSpan.FromSeconds(30), ct);
        }
    }

    public async Task RunOneCycleAsync(CancellationToken ct)
    {
        var lease = _lease.Current;
        var rows = await _queue.ClaimBatchAsync(lease.WorkerEpoch, lease.FencingToken, batchSize: 50, ct);
        foreach (var row in rows)
        {
            if (row.HindsightStatus != SinkStatus.Ok)
                await ProcessSink("hindsight", row, lease, ct);
            if (row.ShodhStatus != SinkStatus.Ok)
                await ProcessSink("shodh", row, lease, ct);
            await ResolveTerminalState(row.EventId, ct);
        }
    }
    // ProcessSink: Client.SendAsync, dann UpdateSinkStatus mit Backoff-Berechnung
    // ResolveTerminalState: Re-Read der Row, entscheidet Complete vs DLQ vs ReleaseLease
}
```
